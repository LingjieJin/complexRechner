00429 static void _win_sndInitRecording(SndDevice_t *psd, int iNewBufferSize, int bNewBlocking)
00430 {
00431     /* Check if device must be opened or reinitialized, default is bNewBlocking=TRUE */
00432     int i;
00433     WaveInOut_t *wpt;
00434     wpt = (WaveInOut_t *)psd->pt; /* convert to correct pointer type */
00435 
00436     DebugCode(printf("*** _win_sndInitRecording\n"););
00437 
00438     if (wpt->bChangDevIn == TRUE)
00439     {   _OpenInDevice(psd);
00440         wpt->bChangDevIn = FALSE; /* Reset flag */
00441     }
00442     wpt->iBufferSizeIn = iNewBufferSize;
00443     wpt->bBlockingRec = bNewBlocking;
00444 
00445     /* Reset interface so that all buffers are returned from the interface */
00446     waveInReset(wpt->m_WaveIn);
00447     waveInStop(wpt->m_WaveIn);
00448 
00449     /* Reset current buffer ID (it is important to do this BEFORE calling */
00450     /* "AddInBuffer()" */
00451     wpt->iWhichBufferIn = 0;
00452 
00453     /* Create memory for sound card buffer */
00454     for (i = 0; i < NUM_SOUND_BUFFERS_IN; i++)
00455     {
00456         /* Unprepare old wave-header in case that we "re-initialized" this */
00457         /* module. Calling "waveInUnprepareHeader()" with an unprepared */
00458         /* buffer (when the module is initialized for the first time) has */
00459         /* simply no effect */
00460         waveInUnprepareHeader(wpt->m_WaveIn, &(wpt->m_WaveInHeader[i]), sizeof(WAVEHDR));
00461 
00462         if (wpt->psSoundcardBuffer[i] != NULL)
00463             free(wpt->psSoundcardBuffer[i]);
00464 
00465         wpt->psSoundcardBuffer[i] = (short *) malloc(sizeof(short)*(wpt->iBufferSizeIn));
00466         _MyAssert(wpt->psSoundcardBuffer[i] != NULL,"malloc crashed");
00467 
00468         /* Send all buffers to driver for filling the queue ----------------- */
00469         /* Prepare buffers before sending them to the sound interface */
00470         _PrepareInBuffer(psd,i);
00471         _AddInBuffer(psd);
00472     }
00473 
00474     /* Notify that sound capturing can start now */
00475     waveInStart(wpt->m_WaveIn);
00476 
00477     /* This reset event is very important for initialization, otherwise we will get errors! */
00478     ResetEvent(wpt->m_WaveInEvent);
00479     DebugCode(printf("--- _win_sndInitRecording\n"););
00480 }

00555     if (wpt->bChangDevIn == TRUE)

00560 
00561         /* Reset flag */
00562         wpt->bChangDevIn = FALSE;
00563     }
00564

00636         /* Reset flag */
00637         wpt->bChangDevOut = FALSE;
00638     }
00639 
00640     /* Get number of "done"-buffers and position of one of them */
00641     _GetDoneBuffer(psd, &iCntPrepBuf, &iIndexDoneBuf);
00642     DebugCode(printf("   iCntPrepBuf:%d  iIndexDoneBuf:%d\n",iCntPrepBuf,iIndexDoneBuf););
00643     /* Now check special cases (Buffer is full or empty) */
00644     if (iCntPrepBuf == 0)
00645     {
00646         if (wpt->bBlockingPlay == TRUE)
00647         {   DebugCode(printf("   bBlockingPlay...\n"););
00648             /* Blocking wave out routine. Needed for transmitter. Always */
00649             /* ensure that the buffer is completely filled to avoid buffer */
00650             /* underruns */
00651             while (iCntPrepBuf == 0)
00652             {   DebugCode(printf("   WaitForSingleObject\n"););
00653                 WaitForSingleObject(wpt->m_WaveOutEvent, INFINITE);
00654                 _GetDoneBuffer(psd, &iCntPrepBuf, &iIndexDoneBuf);
00655             }
00656         }
00657         else
00658         {
00659             /* All buffers are filled, dump new block ----------------------- */
00660             /* It would be better to kill half of the buffer blocks to set the start */

00666     else if (iCntPrepBuf == NUM_SOUND_BUFFERS_OUT)

00670         /* interface until half of the buffer size is reached */
00671         /* Send half of the buffer size blocks to the sound-interface */
00672         DebugCode(printf("   Buffer is empty -> send as many cleared blocks to the sound\n"););
00673         for (j = 0; j < NUM_SOUND_BUFFERS_OUT / 2; j++)
00674         {
00675             /* First, clear these buffers */

00678             /* Then send them to the interface */
00679             _AddOutBuffer(psd, j);
00680         }
00681

00683         iIndexDoneBuf = NUM_SOUND_BUFFERS_OUT / 2;

00689     }
00690

00692     for (i = 0; i < wpt->iBufferSizeOut; i++)
00693         wpt->psPlaybackBuffer[iIndexDoneBuf][i] = psData[i];
00694 
00695     /* Now, send the current block */
00696     _AddOutBuffer(psd, iIndexDoneBuf);
00697 
00698     DebugCode(printf("--- _win_sndWrite\n"););
00699     return bError;
00700 }
